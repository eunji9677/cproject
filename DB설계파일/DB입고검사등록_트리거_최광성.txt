CREATE OR REPLACE TRIGGER TRG_PMM_PM_RIP_LIST
BEFORE
UPDATE OR DELETE ON PMM_PM_RIP_LIST
BEGIN
IF UPDATING THEN
    RAISE_APPLICATION_ERROR(-20000,'입고검사등록 정보 수정불가');
ELSIF DELETING THEN
    RAISE_APPLICATION_ERROR(-20001,'입고검사등록 정보 삭제불가');
ELSE
    RAISE_APPLICATION_ERROR(-20002,'입고검사등록 정보 변경불가');
END IF;
END;
